class MyList(list):

    def __setitem__(self, index, value):
        if index == 0:
            raise IndexError
        if index > 0:
            index -= 1
        super().__setitem__(index, value)

    def __getitem__(self, index):
        if index == 0:
            raise IndexError
        if index > 0:
            index -= 1
        return super().__getitem__(index)


mylist = MyList(['a', 'b', 'c', 'd', 'e'])

mylist[1] = 'apple'
mylist[5] = 'orange'
